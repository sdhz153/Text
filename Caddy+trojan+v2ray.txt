三、安裝Caddy
安裝主文件
curl https://getcaddy.com | bash -s personal

配置 Caddy
文件目錄
mkdir /etc/caddy

chown -R root:root /etc/caddy

mkdir /etc/ssl/caddy

chown -R root:www-data /etc/ssl/caddy

                                  #(770)      只有拥有者，属组用户有读写執行权限；其他用户無权限。

chmod 770 /etc/ssl/caddy


配置文件填充内容
touch /etc/caddy/Caddyfile

nano /etc/caddy/Caddyfile

=============================================================================================================

:80 {
  proxy / https://www.china.com {
    except /350b8257-b6ff-ad55-3a10-83ba6e3e3c38/
  }
  timeouts none
  gzip
  proxy /350b8257-b6ff-ad55-3a10-83ba6e3e3c38/ 127.0.0.1:40754 {
  websocket
  header_upstream -Origin
  }
}

================================================================================================================

chown root:root /etc/caddy/Caddyfile

                #(644)      只有拥有者有读写权限；而属组用户和其他用户只有读权限。

chmod 644 /etc/caddy/Caddyfile


系統服務
wget https://raw.githubusercontent.com/caddyserver/caddy/v1/dist/init/linux-systemd/caddy.service

cp caddy.service /etc/systemd/system/

chown root:root /etc/systemd/system/caddy.service

         #(644)      只有拥有者有读写权限；而属组用户和其他用户只有读权限。

chmod 644 /etc/systemd/system/caddy.service

systemctl daemon-reload

啓動

systemctl start caddy

systemctl status caddy

systemctl enable caddy #设置为开机自动启动

systemctl restart caddy   #每次修改后都要执行一次重启


#停止caddy

service caddy stop

六、加速優化
BBR
修改系统变量
echo "net.core.default_qdisc=fq" >> /etc/sysctl.conf
echo "net.ipv4.tcp_congestion_control=bbr" >> /etc/sysctl.conf
保存生效
sysctl -p
查看内核是否已开启BBR

lsmod | grep bbr

返回值有 tcp_bbr 模块即说明 bbr 已启动

TLS开启OSCP

openssl s_client -connect www.xxxxxx.tk:443 -status -tlsextdebug < /dev/null 2>&1 | grep -i "OCSP response"


TCP开启 fastopen

echo 3 > /proc/sys/net/ipv4/tcp_fastopen


安裝 UFW和編輯器nano

apt-get install ufw nano

啓用UFW

ufw enable


設置初始默認狀態

ufw default deny incoming

ufw default allow outgoing

開放常用端口

ufw allow ssh

ufw allow http

ufw allow https

ufw allow 40754/udp    (v2ray的config.json下的"port": 40754,)

重啓UFW使設置生效

ufw reload


查看UFW狀態

ufw status verbose


啓用日志，默認級別low

ufw logging on

查看日志

nano /var/log/ufw.log

其他指令
重置UFW配置
ufw reset

查看UFW配置文件
 nano /etc/default/ufw
